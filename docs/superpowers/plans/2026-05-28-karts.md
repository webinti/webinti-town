# F11 — Karts Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add 5 ride-able karts that spawn at a parking near the map entrance, let players move at x2 speed (x3 during a 2 s boost with 15 s cooldown), push other players on collision, and auto-return to parking after 5 min of inactivity.

**Architecture:** Server-authoritative state via `KartManager` (mirrors the existing `WorkstationManager` pattern, no persistence). Karts and `PlayerState.kartId` are broadcast through Socket.IO; the client mirrors them in `gameStore` and renders procedural Phaser Graphics sprites via a new `KartOverlay`. Player speed is derived from `(kartId, boosting)` through a pure helper so it's trivially testable.

**Tech Stack:** TypeScript, Vitest, Node, Socket.IO, Phaser 3, Zustand, React. Backend `server/`, frontend `client/`. Code conventions follow existing F6 (workstations) implementation.

**Spec:** `docs/superpowers/specs/2026-05-28-karts-design.md`

```ts
import type { KartDef } from '../karts.js';

export interface KartState {
  id: string;
  x: number;
  y: number;
  parkingX: number;
  parkingY: number;
  driverId: string | null;
  lastMovedAt: number;
}

export class KartManager {
  private readonly defs: readonly KartDef[];
  private readonly states = new Map<string, KartState>();
  private readonly now: () => number;

  constructor(defs: readonly KartDef[], now: () => number = Date.now) {
    this.defs = defs;
    this.now = now;
    const t = now();
    for (const d of defs) {
      this.states.set(d.id, {
        id: d.id,
        x: d.parkingX,
        y: d.parkingY,
        parkingX: d.parkingX,
        parkingY: d.parkingY,
        driverId: null,
        lastMovedAt: t,
      });
    }
  }

  getState(id: string): KartState | undefined {
    return this.states.get(id);
  }

  getAllStates(): KartState[] {
    return this.defs.map((d) => ({ ...this.states.get(d.id)! }));
  }
}
```

```ts
  mount(kartId: string, playerId: string, playerX: number, playerY: number): boolean {
    const k = this.states.get(kartId);
    if (!k) return false;
    if (k.driverId !== null) return false;
    const dx = playerX - k.x;
    const dy = playerY - k.y;
    if (Math.hypot(dx, dy) > MOUNT_DISTANCE) return false;
    k.driverId = playerId;
    k.lastMovedAt = this.now();
    return true;
  }

  getKartByDriver(playerId: string): KartState | undefined {
    for (const k of this.states.values()) {
      if (k.driverId === playerId) return k;
    }
    return undefined;
  }
```

```ts
  /**
   * Repositionne au parking les karts libres + immobiles depuis > KART_IDLE_RETURN_MS.
   * Retourne la liste des ids déplacés (pour broadcast côté handler).
   */
  sweepIdle(): string[] {
    const now = this.now();
    const moved: string[] = [];
    for (const k of this.states.values()) {
      if (k.driverId !== null) continue;
      if (k.x === k.parkingX && k.y === k.parkingY) continue;
      if (now - k.lastMovedAt <= KART_IDLE_RETURN_MS) continue;
      k.x = k.parkingX;
      k.y = k.parkingY;
      k.lastMovedAt = now;
      moved.push(k.id);
    }
    return moved;
  }
```

```ts
import Phaser from 'phaser';
import type { KartState } from '../types';

const COLOR_BODY = 0xfacc15;     // jaune kart
const COLOR_EDGE = 0x000000;
const COLOR_WHEEL = 0x111111;
const KART_W = 28;
const KART_H = 20;

export class KartOverlay {
  private readonly gfx: Phaser.GameObjects.Graphics;

  constructor(scene: Phaser.Scene) {
    // Depth 8 : entre le sol (0) et les sprites joueurs (9).
    this.gfx = scene.add.graphics().setDepth(8);
  }

  update(karts: Map<string, KartState>): void {
    this.gfx.clear();
    for (const k of karts.values()) {
      const x = Math.round(k.x);
      const y = Math.round(k.y);
      // Corps
      this.gfx.fillStyle(COLOR_BODY, 1);
      this.gfx.fillRect(x - KART_W / 2, y - KART_H / 2, KART_W, KART_H);
      this.gfx.lineStyle(1, COLOR_EDGE, 1);
      this.gfx.strokeRect(x - KART_W / 2, y - KART_H / 2, KART_W, KART_H);
      // Roues (4 petits cercles)
      this.gfx.fillStyle(COLOR_WHEEL, 1);
      this.gfx.fillCircle(x - KART_W / 2 + 3, y - KART_H / 2 + 3, 2);
      this.gfx.fillCircle(x + KART_W / 2 - 3, y - KART_H / 2 + 3, 2);
      this.gfx.fillCircle(x - KART_W / 2 + 3, y + KART_H / 2 - 3, 2);
      this.gfx.fillCircle(x + KART_W / 2 - 3, y + KART_H / 2 - 3, 2);
      // Petit triangle "avant" — pointe vers le haut par défaut
      this.gfx.fillStyle(COLOR_EDGE, 1);
      this.gfx.fillTriangle(x, y - KART_H / 2 - 3, x - 3, y - KART_H / 2 + 1, x + 3, y - KART_H / 2 + 1);
    }
  }

  destroy(): void {
    this.gfx.destroy();
  }
}
```
